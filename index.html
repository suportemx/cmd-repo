script_ping #!/bin/bash

if [ "$1" == "" ]
then
echo "Ping Sweep"
echo "Modo de uso: $0 REDE"
echo "Ex: $0 192.168.1"
else
for host in {1..254};
do
ping -c 1 $1.$host | grep "64 bytes"| cut -d " " -f 4 | sed 's/.$//';
done
fi
 Script ping
script_portscan #!/bin/bash
if [ $1 == "" ]
then
echo "PortScan Network"
echo "Modo de uso: $ REDE PORTA"
echo "Ex: $0 192.168.1 80"
else
for ip in {1..254};
do
sudo hping3 -S -p $2 -c 1 $1.$ip 2> /dev/null | grep "flags=SA" | cut -d " " -f2 | cut -d "=" -f2;
done
fi
